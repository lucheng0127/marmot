## 3.2 BPF Map Schema

### 3.2.1 Flow BPF MapпјҲж ёеҝғпјү

```c
// ж–Үд»¶: bpf/maps.h

// 5-tuple жөҒй”®
struct flow_key {
    __u32 src_ip;
    __u32 dst_ip;
    __u16 src_port;
    __u16 dst_port;
    __u8  protocol;   // IPPROTO_TCP=6, IPPROTO_UDP=17
    __u8  pad;        // еҜ№йҪҗеЎ«е……
};

// жөҒзј“еӯҳеҖј
struct flow_value {
    __u8  action;         // FLOW_ACTION_DIRECT=0
                          // FLOW_ACTION_PROXY=1
                          // FLOW_ACTION_BLOCK=2
    __u8  outbound_tag[16]; // proxy outbound tag (еҰӮ "proxy-jp\0")
    __u32 expire_at;      // иҝҮжңҹж—¶й—ҙ (bpf_ktime_get_ns / 1000000000 + ttl)
    __u32 hit_count;      // е‘Ҫдёӯж¬Ўж•°пјҲз»ҹи®Ўз”Ёпјү
    __u8  reserved[4];    // йў„з•ҷпјҲеҜ№йҪҗеҲ° 32 еӯ—иҠӮпјү
};

#define FLOW_MAP_MAX_ENTRIES 65536
#define FLOW_TTL_TCP  3600   // TCP: 1h
#define FLOW_TTL_UDP  120    // UDP: 2min

struct {
    __uint(type, BPF_MAP_TYPE_HASH);
    __uint(max_entries, FLOW_MAP_MAX_ENTRIES);
    __type(key, struct flow_key);
    __type(value, struct flow_value);
} flow_map SEC(".maps");
```

```go
// FlowKey вҖ” 5-tuple иҝһжҺҘж ҮиҜҶ
type FlowKey struct {
    SrcIP    net.IP `json:"src_ip"`
    DstIP    net.IP `json:"dst_ip"`
    SrcPort  uint16 `json:"src_port"`
    DstPort  uint16 `json:"dst_port"`
    Protocol uint8  `json:"protocol"` // 6=TCP, 17=UDP
}

// FlowEntry вҖ” з”ЁжҲ·жҖҒзј“еӯҳжқЎзӣ®

    Key        FlowKey   `json:"key"`
    Action     Action    `json:"action"`
    RuleID     string    `json:"rule_id"`
    RuleType   string    `json:"rule_type"`
    CreatedAt  time.Time `json:"created_at"`

// Insert жҸ’е…Ҙзј“еӯҳ + еҗҢжӯҘеҲ° BPF Map
func (c *Cache) Insert(key FlowKey, entry *FlowEntry) error {
```

### 3.9.4 Go вҶ’ eBPF SDK д»Јз ҒзӨәдҫӢ

```go
// pkg/bpf/loader.go вҖ” eBPF еҠ иҪҪдёҺ Map ж“ҚдҪң

import (
    "github.com/cilium/ebpf"        // eBPF еҠ иҪҪ
    "github.com/cilium/ebpf/link"   // зЁӢеәҸжҢӮиҪҪ
    "github.com/cilium/ebpf/rlimit" // иө„жәҗйҷҗеҲ¶
)

// иҮӘеҠЁз”ҹжҲҗзҡ„ Go з»“жһ„дҪ“пјҲеҜ№еә” maps.hпјү
//go:generate go run github.com/cilium/ebpf/cmd/bpf2go -cc clang bpf ./bpf/tc_ingress.c

type FlowCacheManager struct {
    objs     bpfObjects        // eBPF зј–иҜ‘дә§зү© Go binding
    flowMap  *ebpf.Map         // Flow BPF Map еҸҘжҹ„
    cidrMap  *ebpf.Map         // CIDR зҷҪеҗҚеҚ• Map еҸҘжҹ„
    cache    *conntrack.Cache  // з”ЁжҲ·жҖҒ Conntrack Cache
}

func (m *FlowCacheManager) InsertFlow(key FlowKey, action Action, outboundTag string) error {
    var k bpfFlowKey
    var v bpfFlowValue

    // еЎ«е…… key
    k.SrcIp = nativeEndian.Uint32(key.SrcIP.To4())
    k.DstIp = nativeEndian.Uint32(key.DstIP.To4())
    k.SrcPort = htons(key.SrcPort)
    k.DstPort = htons(key.DstPort)
    k.Protocol = key.Protocol

    // еЎ«е…… value
    v.Action = actionCode(action.Mode)    // 0=direct, 1=proxy, 2=block
    copy(v.OutboundTag[:], outboundTag)
    v.ExpireAt = uint32(time.Now().Unix() + flowTTL)

    // еҶҷе…Ҙ BPF Map
    return m.flowMap.Put(&k, &v, ebpf.UpdateAny)
}

func (m *FlowCacheManager) DeleteFlow(key FlowKey) error {
    var k bpfFlowKey
    // ... еЎ«е…… key еҗҢдёҠ
    return m.flowMap.Delete(&k)
}

func (m *FlowCacheManager) GC(ctx context.Context) {
    ticker := time.NewTicker(30 * time.Second)
    for {
        select {
        case <-ctx.Done():
            return
        case <-ticker.C:
            m.cache.Evict() // жё…зҗҶиҝҮжңҹжқЎзӣ®
        }
    }
}
```
